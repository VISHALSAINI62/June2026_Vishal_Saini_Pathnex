    Ansible -- Install Docker
 
  - name: Install Docker
    hosts: all
    become: yes
    tasks:
      - name: Install Docker
        yum:
          name: docker
          state: present
      - name: Start Docker
        service:
          name: docker
          state: started
          enabled: yes

     Kubernetes - Pod with Volume Mount
  apiVersion: v1
  kind: pod
  metadata:
    name: pathnex-pod
  spec:
    containers:
      - name: web
        image: nginx
        volumeMounts:
          -mountpath: /usr/share/nginx/html
           name: html-volume
     volumes:
       - name: html-volume
        hostpath:
          path: /data
          type: Directory
       Shell Scripts - Check for Running Services
    #!/bin/bash
    services=('nginx' 'docker' 'httpd')
    for service in '${services[@]}'
    do
      if systemctl is-active --quite $services; then
        echo '$service is running'
      else
        echo '$service is not running'
      fi
    done
